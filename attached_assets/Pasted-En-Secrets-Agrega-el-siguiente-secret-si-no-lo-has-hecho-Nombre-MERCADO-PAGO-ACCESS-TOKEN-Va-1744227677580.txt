const express = require('express');
const mercadopago = require('mercadopago');

const app = express();
app.use(express.json());

// Configurar Mercado Pago con el Access Token de prueba
mercadopago.configure({ access_token: process.env.MERCADO_PAGO_ACCESS_TOKEN });

// Endpoint para generar enlace de pago con Mercado Pago Checkout Pro
app.post('/generar-enlace', async (req, res) => {
  try {
    const { cuotas } = req.body;
    console.log('Cuotas recibidas:', cuotas);

    // Calcular el monto total (asumimos que 'total' está en centavos)
    const montoTotal = cuotas.reduce((sum, cuota) => sum + cuota.total, 0) / 100;
    console.log('Monto total calculado:', montoTotal);

    // Configurar la preferencia de pago
    const preference = {
      items: [
        {
          title: "Pago de Cuotas",
          quantity: 1,
          unit_price: montoTotal,
          currency_id: "CLP"
        }
      ],
      back_urls: {
        success: "https://98856977-dfac-434c-829e-a30c7324196b-00-2aqs2tqhqe1if.riker.replit.dev/success",
        failure: "https://98856977-dfac-434c-829e-a30c7324196b-00-2aqs2tqhqe1if.riker.replit.dev/failure",
        pending: "https://98856977-dfac-434c-829e-a30c7324196b-00-2aqs2tqhqe1if.riker.replit.dev/pending"
      }
      // Nota: Se elimina el auto_return para evitar redirecciones prematuras.
    };
    console.log('Preferencia enviada:', preference);

    // Crear la preferencia en Mercado Pago
    const response = await mercadopago.preferences.create(preference);
    console.log('Respuesta de Mercado Pago:', response.body);

    res.json({ paymentLink: response.body.init_point });
  } catch (error) {
    console.error('Error al generar el enlace:', error);
    res.status(500).json({ error: 'Error al generar el enlace' });
  }
});

app.listen(3000, () => console.log('Servidor en puerto 3000'));

async function continuarPago() {
  const totalPagar = document.getElementById('totalPagar').textContent;
  if (totalPagar === '$0') {
    alert("Por favor seleccione al menos una cuota para continuar.");
    return;
  }
  const cuotasSeleccionadas = [];
  document.querySelectorAll('#cuotasList .cuota-item').forEach((item, index) => {
    const checkbox = item.querySelector('input[type="checkbox"]');
    if (checkbox.checked) {
      // Extraemos el valor del Total Cuota; se asume que está en el formato "$1.359.265"
      const totalText = item.querySelector('.cuota-details p:last-child').textContent.split(': ')[1];
      const total = parseInt(totalText.replace('$', '').replace(/\./g, ''));
      cuotasSeleccionadas.push({ quantity: 1, total });
    }
  });

  try {
    const response = await fetch('https://98856977-dfac-434c-829e-a30c7324196b-00-2aqs2tqhqe1if.riker.replit.dev/generar-enlace', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ cuotas: cuotasSeleccionadas })
    });
    const data = await response.json();
    console.log('Respuesta del backend:', data);
    if (data.paymentLink) {
      console.log('Redirigiendo a:', data.paymentLink);
      window.location.href = data.paymentLink;
    } else {
      alert("No se recibió un enlace de pago válido: " + JSON.stringify(data));
    }
  } catch (error) {
    alert("Error al generar el enlace de pago");
  }
}
